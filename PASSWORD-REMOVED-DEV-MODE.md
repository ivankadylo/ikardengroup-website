## 🔧 ЩО БУЛО ЗМІНЕНО В КОДІ:

### **Файл: js/script.js**

**Функція showAdminLogin():**

```javascript
// СТАРИЙ КОД (з паролем):
function showAdminLogin() {
    // Показував модальне вікно
    // З полями логін/пароль
    // Потрібно було вводити
}

// НОВИЙ КОД (БЕЗ пароля):
function showAdminLogin() {
    // Перевірка чи вже в admin
    const isLoggedIn = localStorage.getItem('ikarden-admin-logged-in');
    
    if (isLoggedIn === 'true') {
        toggleAdminMode();
        return;
    }
    
    // ОДРАЗУ активуємо admin!
    localStorage.setItem('ikarden-admin-logged-in', 'true');
    toggleAdminMode();
    showNotification('🔓 Admin режим активовано');
}
```
